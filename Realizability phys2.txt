\begin{abstract}
Zeno's paradox implicitly assumes that spatial displacements carry irreducible positive
cost at every scale. We negate this: if spatial displacement sequences exist whose
cost-to-magnitude ratio vanishes (Assumption~R), while pure temporal displacements
retain positive cost (Assumption~T), then in $(1{+}1)$ dimensions any quadratic
leading-order expansion of the cost function is indefinite---provided the quadratic
form is nondegenerate, an explicit condition that is separately required.
After normalisation,
\[
Q = dt^{2} - c_{\max}^{-2}\,dr^{2}.
\]
Lorentzian signature therefore emerges from the asymmetry between temporal and spatial
displacement costs rather than from any geometric postulate. Every non-Lorentzian alternative is excluded: positive-definite
by Assumption~R together with the Lemma; negative-definite by Assumption~T;
and degenerate by the nondegeneracy hypothesis of Theorem~\ref{thm:lorentz}. In $(1{+}3)$~dimensions, under a
strengthened three-directional zero-threshold condition together with nondegeneracy
of~$Q$, the only possible signature is $\mathrm{Sig}(Q)=(1,3)$.

The present derivation differs from the EPS axiomatisation~\cite{EPS} in its starting
point: no particles, light signals, or kinematic objects are presupposed---only a
scalar cost function. This provides a more primitive geometric foundation for the
thermodynamic derivation of Einstein's equations~\cite{Jacobson}: Lorentzian signature
itself is derived rather than assumed.
\end{abstract}

\section{Introduction}

We show that a Lorentzian causal structure follows from a minimal asymmetry between
temporal and spatial displacement costs together with a quadratic leading-order
expansion. The closest precursor is the EPS axiomatisation~\cite{EPS}; unlike EPS,
no particles, light signals, or worldlines appear as primitives --- only a scalar
cost function.

The title warrants a brief clarification. By \textit{origin of causality} we mean a
logical rather than a physical origin: a minimal set of conditions on a cost function
from which Lorentzian causal structure follows as the unique algebraic consequence.

\medskip
\noindent\textbf{Main Theorem (informal).}
Under Assumptions~R, E, T, and nondegeneracy of~$Q$, the leading quadratic
form~$Q$ is Lorentzian in $(1{+}1)$ dimensions:
\[
Q = dt^{2} - c_{\max}^{-2}\,dr^{2}, \qquad c_{\max}>0,
\]
up to normalisation and an invertible linear change of variables. In $(1{+}3)$
dimensions, under R4, E4, T, and nondegeneracy of~$Q$, the only possible signature
is $\mathrm{Sig}(Q)=(1,3)$.

The precise setup, statements, and proofs are given in Sections~2--4.

\section{Setup}

\noindent\textbf{Notation.}
$d(x;\delta x)\ge 0$: displacement cost at~$x$ for displacement~$\delta x$.
$Q$: the leading-order quadratic form of~$d^{2}$ (Assumption~E).
$\mathrm{Sig}(Q)=(p,q)$: inertia of~$Q$ ($p$~positive, $q$~negative eigenvalues).
$(u)_{+}:=\max\{u,0\}$: positive part.
$c_{\max}>0$: maximal propagation speed encoded in~$Q$.
$c_{t}>0$: lower bound on temporal cost per unit displacement (Assumption~T).
$\|\cdot\|$: any norm on~$\mathbb{R}^{2}$; conclusions are norm-independent.
$S^{1}=\{u\in\mathbb{R}^{2}:\|u\|=1\}$: unit circle.

\medskip
Let $U\subset\mathbb{R}^{2}$ be open with chart $(t,r)$, and fix $x\in U$.

\begin{definition}[R --- Zero-threshold realizability]\label{def:R}
There exists a sequence $\delta x_{n}=(\delta t_{n},\delta r_{n})\to 0$ with
$\delta t_{n}>0$, $\delta r_{n}\neq 0$ such that
$\frac{d(x;\delta x_{n})}{\|\delta x_{n}\|}\longrightarrow 0$,
and a subsequence $v_{n_{k}}\to v=(v_{t},v_{r})$ with $v_{t}>0$ and $v_{r}\neq 0$.
\end{definition}

\begin{definition}[E --- Quadratic expansion]\label{def:E}
There is a quadratic form $Q(\delta t,\delta r)=a\,\delta t^{2}+2\kappa\,\delta t\,\delta r+b\,\delta r^{2}$
such that $d(x;\delta x)^{2} = \bigl(Q(\delta x)\bigr)_{+} + o\!\left(\|\delta x\|^{2}\right)$.
\end{definition}

\begin{definition}[T --- Temporal cost]\label{def:T}
There exists $c_{t}>0$ such that for all sufficiently small $\Delta t>0$,
$d\bigl(x;(\Delta t,0)\bigr)\ge c_{t}\,\Delta t$.
\end{definition}

\noindent The three assumptions are summarised as follows:
Definition~\ref{def:R} (Assumption~R) provides a zero-cost spatial direction;
Definition~\ref{def:E} (Assumption~E) establishes the quadratic leading-order structure;
Definition~\ref{def:T} (Assumption~T) fixes the positive cost of pure temporal displacement.

\section{Main result in $(1{+}1)$ dimensions}

\begin{lemma}[Null--Sign equivalence]\label{lem:null}
Assume~E. If $\delta x_{n}\to 0$ with $\delta t_{n}>0$,
$d(x;\delta x_{n})/\|\delta x_{n}\|\to 0$, and a subsequence converges to
$v\in S^{1}$, then $Q(v)\le 0$. Conversely, if $Q(v)\le 0$ and
$\delta x_{n}/\|\delta x_{n}\|\to v$, then $d(x;\delta x_{n})/\|\delta x_{n}\|\to 0$.
\end{lemma}

\begin{proof}
\textit{Forward:} $(d/\|\delta x_{n_{k}}\|)^{2}=(Q(v_{n_{k}}))_{+}+o(1)\to 0$,
so $(Q(v))_{+}=0$.
\textit{Converse:} $(Q(v_{n}))_{+}\to 0$, so
$(Q(\delta x_{n}))_{+}=\|\delta x_{n}\|^{2}(Q(v_{n}))_{+}=o(\|\delta x_{n}\|^{2})$,
giving $d/\|\delta x_{n}\|\to 0$.\qed
\end{proof}

\begin{theorem}[Lorentzian signature from realizability]\label{thm:lorentz}
Assume R, E, T, and that $Q$ is nondegenerate.
Then $Q$ is indefinite, and after rescaling~$t$ so that $Q(1,0)=1$
and applying an invertible linear change of variables,
$Q = dt^{2} - c_{\max}^{-2}\,dr^{2}$ for some $c_{\max}>0$.
The null set consists of exactly two directions $dr/dt=\pm c_{\max}$.
\end{theorem}

\begin{proof}
\textbf{Step~1.} By~R and Lemma~\ref{lem:null}, $Q(v)\le 0$.

\textbf{Step~2.} $Q(1,0)>0$ by~T.

\textbf{Step~3.} Define
\[
\gamma(s):=\frac{(1-s)(1,0)+s\,v}{\|(1-s)(1,0)+s\,v\|},\qquad s\in[0,1].
\]
Since $v_{r}\neq 0$ (Assumption~R), the denominator is nonzero for all $s\in[0,1]$,
so $\gamma$ is well-defined and continuous. Since $v_{t}>0$, the numerator has
strictly positive $t$-component throughout, so $\gamma$ is a path in
$S^{1}\cap\{u_{t}>0\}$. Then $Q(\gamma(0))=Q(1,0)>0$ (Step~2) and
$Q(\gamma(1))=Q(v/\|v\|)\le 0$ (Step~1), so by the intermediate value theorem
there exists $s^{*}$ with $Q(\gamma(s^{*}))=0$. Set $w:=\gamma(s^{*})$;
then $w_{t}>0$. Since $Q(w)=0$ but $Q(1,0)>0$, $w\neq(1,0)$, so $w_{r}\neq 0$.

\textbf{Step~4 (Signature and normalisation).}
By Step~2, $Q$ takes a positive value; by Step~3, $Q$ vanishes on a nonzero
vector~$w$ with $w_{t}>0$. Since $Q$ is nondegenerate (hypothesis), a nondegenerate
quadratic form on~$\mathbb{R}^{2}$ that takes both positive values and zero on a
nonzero vector cannot be definite; hence $Q$ is indefinite and
$\mathrm{Sig}(Q)=(1,1)$. Rescaling~$t$ so that $Q(1,0)=1$ and applying an
invertible linear change of variables gives
$Q=dt^{2}-c_{\max}^{-2}\,dr^{2}$.\qedhere
\end{proof}

\begin{remark}[The speed $c_{\max}$]
$c_{\max}$ arises from the cost structure, not as an independent postulate.
\end{remark}

\begin{remark}[Physical interpretation of $d$]
The derivation requires only R, E, and T, not a specific physical origin.
\end{remark}

\section{Extension to $(1{+}3)$ dimensions}

Definition~\ref{def:R4} (Assumption~R4) strengthens the two-dimensional
Assumption~R to three independent spatial directions.
Definition~\ref{def:E4} (Assumption~E4) extends the quadratic expansion to~$\mathbb{R}^{4}$.

\begin{definition}[R4]\label{def:R4}
For each $i\in\{1,2,3\}$ there exists a sequence converging to $v^{(i)}$ with
$v_{t}^{(i)}>0$ and $v_{i}^{(i)}\neq 0$. The three limiting directions span a
subspace~$W$ with $Q|_{W}$ negative semidefinite.
\end{definition}

\begin{definition}[E4]\label{def:E4}
There is a quadratic form $Q$ on~$\mathbb{R}^{4}$ with
$d(x;\delta x)^{2} = \bigl(Q(\delta x)\bigr)_{+} + o\!\left(\|\delta x\|^{2}\right)$.
\end{definition}

\begin{theorem}[Lorentzian signature --- $(1{+}3)$D]\label{thm:4d}
Assume R4, E4, T, and nondegeneracy of~$Q$. Then $\mathrm{Sig}(Q)=(1,3)$.
\end{theorem}

\begin{proof}
T gives positive index $\ge 1$. R4 gives negative index $\ge 3$. Nondegeneracy
forces sum~$=4$, so $\mathrm{Sig}(Q)=(1,3)$.\qed
\end{proof}

\begin{remark}[Nondegeneracy]
Both Theorem~\ref{thm:lorentz} and Theorem~\ref{thm:4d} take nondegeneracy
of~$Q$ as an explicit hypothesis.
In $(1{+}1)$ dimensions, ruling out degeneracy from Assumptions~R, E, T alone
is not possible: the degenerate form $Q=(dt-\lambda\,dr)^{2}$ satisfies all
three assumptions for any $\lambda\neq 0$ (choosing the sign so that the null
direction has $v_{t}>0$).
Nondegeneracy is therefore an explicit scope condition in both theorems,
not a hidden assumption.
\end{remark}

\begin{remark}[Exclusion of non-Lorentzian signatures]
(i)~Positive-definite excluded by R and Lemma~\ref{lem:null}.
(ii)~Negative-definite excluded by T.
(iii)~Degenerate forms are excluded by the nondegeneracy hypothesis of
Theorem~\ref{thm:lorentz}: under that hypothesis, a nonzero null vector
(Step~3) forces the form to be indefinite rather than degenerate.
(iv)~Split signature $(2,2)$ excluded by R4.
\end{remark}
