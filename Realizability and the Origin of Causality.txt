\title{\textbf{Realizability and the Origin of Causality}}
\author{Y.\,Y.\,N.\ Li}
\date{}

\begin{abstract}
Zeno's paradox implicitly assumes that spatial displacements carry irreducible positive
cost at every scale. We negate this: if spatial displacement sequences exist whose cost-to-magnitude ratio vanishes (Assumption~R), while pure temporal displacements retain positive
cost (Assumption~T), then in $(1{+}1)$ dimensions any quadratic leading-order expansion of the
cost function is forced to be nondegenerate and indefinite. After normalisation,
\[
Q = dt^{2} - c_{\max}^{-2}\,dr^{2}.
\]
Lorentzian signature therefore emerges from the asymmetry between temporal and spatial
displacement costs rather than from any geometric postulate. In $(1{+}3)$ dimensions, under a
strengthened three-directional zero-threshold condition together with nondegeneracy of~$Q$,
the only possible signature is $\mathrm{Sig}(Q)=(1,3)$.

The present derivation differs from the EPS axiomatisation~\cite{EPS} in its starting point: no
particles, light signals, or kinematic objects are presupposed. This provides a more primitive
geometric foundation for the thermodynamic derivation of Einstein's equations~\cite{Jacobson}.
\end{abstract}

\section{Introduction}

We show that a Lorentzian causal structure follows from a minimal asymmetry between temporal and spatial displacement costs together with a quadratic leading-order expansion. This
asymmetry forces the associated quadratic form to be indefinite, yielding Lorentzian causal
structure without assuming any metric or \textit{a priori} causal cone. The closest precursor is the EPS
axiomatisation~\cite{EPS}; unlike EPS, no particles, light signals, or worldlines appear as primitives ---
only a scalar cost function. The result provides a foundation one step earlier than the Jacobson
derivation~\cite{Jacobson}: Lorentzian signature itself is derived rather than assumed.

Zeno's assumption --- that every spatial displacement carries irreducible positive cost --- corresponds precisely to positive-definite signature $\mathrm{Sig}(Q)=(2,0)$.
Assumption~R negates this: it asserts the existence of displacement sequences whose cost-to-magnitude ratio vanishes. This single negation, together with the positive temporal cost~(T), is what forces the signature to be Lorentzian.

The title warrants a brief clarification. By \textit{origin of causality} we mean a logical rather than
a physical origin: a minimal set of conditions on a cost function from which Lorentzian causal
structure follows as the unique algebraic consequence, with no other signature possible. The
present paper identifies this set as Assumptions~R, E, and~T. The physical origin of the cost
asymmetry itself lies outside the scope of the present paper.

\medskip
\noindent\textbf{Main Theorem (informal).}
Under Assumptions~R, E, and~T (and no others), the leading
quadratic form~$Q$ is Lorentzian in $(1{+}1)$ dimensions:
\[
Q = dt^{2} - c_{\max}^{-2}\,dr^{2}, \qquad c_{\max}>0,
\]
up to normalisation and an invertible linear change of variables. In $(1{+}3)$ dimensions, under
R4, E4, T, and nondegeneracy of~$Q$, the only possible signature is $\mathrm{Sig}(Q)=(1,3)$.

The precise setup, statements, and proofs are given in Sections~2--4.

\section{Setup}

Work in an open $U\subset\mathbb{R}^{2}$ with chart $(t,r)$, and fix $x\in U$. For small $\delta x=(\delta t,\delta r)$ with $\delta t>0$, let
$d(x;\delta x)\ge 0$ be a displacement cost at~$x$. The norm $\|\cdot\|$ denotes any norm on~$\mathbb{R}^{2}$; all conclusions
are independent of the choice. The chart variables $t$ and~$r$ are a distinguished direction (carrying
positive cost, Assumption~T) and its complement; no metric or causal structure on~$U$ is assumed
at this stage.

\begin{definition}[R --- Zero-threshold realizability]
There exists a sequence $\delta x_{n}=(\delta t_{n},\delta r_{n})\to 0$
with $\delta t_{n}>0$, $\delta r_{n}\neq 0$ such that
\[
\frac{d(x;\delta x_{n})}{\|\delta x_{n}\|}\longrightarrow 0,
\]
and there exists a subsequence $v_{n_{k}}\to v=(v_{t},v_{r})$ with $v_{t}>0$ and $v_{r}\neq 0$, where $v_{n_{k}}:=\delta x_{n_{k}}/\|\delta x_{n_{k}}\|$.
\end{definition}

Assumption~E formalises the leading-order structure of the cost function within which Assumption~R operates.

\begin{definition}[E --- Quadratic expansion]
There is a quadratic form $Q(\delta t,\delta r)=a\,\delta t^{2}+2\kappa\,\delta t\,\delta r+b\,\delta r^{2}$ such that
\[
d(x;\delta x)^{2} = \bigl(Q(\delta x)\bigr)_{+} + o\!\left(\|\delta x\|^{2}\right) \quad\text{as } \delta x\to 0 \text{ with } \delta t>0,
\]
where $(u)_{+}:=\max\{u,0\}$.
The positive part ensures compatibility with $d\ge 0$: we do not assume $Q\ge 0$, since the
existence of directions on which $Q\le 0$ is to be derived, not postulated.
\end{definition}

The notation $\mathrm{Sig}(Q)=(p,q)$ denotes the inertia of~$Q$: $p$~positive and $q$~negative eigenvalues.

\begin{definition}[T --- Temporal cost]
There exists $c_{t}>0$ such that for all sufficiently small $\Delta t>0$,
\[
d\bigl(x;(\Delta t,0)\bigr)\ge c_{t}\,\Delta t.
\]
\end{definition}

\section{Main result in $(1{+}1)$ dimensions}

\begin{lemma}[Null--Sign equivalence]
Assume~E. If $\delta x_{n}\to 0$ with $\delta t_{n}>0$, $d(x;\delta x_{n})/\|\delta x_{n}\|\to 0$,
and a subsequence of $v_{n}:=\delta x_{n}/\|\delta x_{n}\|$ converges to $v\in S^{1}$, then $Q(v)\le 0$. Conversely, if
$Q(v)\le 0$ and $\delta x_{n}\to 0$ with $\delta x_{n}/\|\delta x_{n}\|\to v$, then $d(x;\delta x_{n})/\|\delta x_{n}\|\to 0$.
\end{lemma}

\begin{proof}
\textit{Forward:} By~E, $(d/\|\delta x_{n_{k}}\|)^{2}=(Q(v_{n_{k}}))_{+}+o(1)\to 0$, so $(Q(v))_{+}=0$.

\textit{Converse:} Let $v_{n}:=\delta x_{n}/\|\delta x_{n}\|$, so $v_{n}\to v$. Since $Q$ is continuous and $Q(v)\le 0$, we have
$(Q(v_{n}))_{+}\to 0$. By homogeneity of~$Q$,
\[
\bigl(Q(\delta x_{n})\bigr)_{+} = \|\delta x_{n}\|^{2}\bigl(Q(v_{n})\bigr)_{+} = o\!\left(\|\delta x_{n}\|^{2}\right).
\]
Hence Assumption~E gives $d(x;\delta x_{n})^{2}=o(\|\delta x_{n}\|^{2})$, so $d(x;\delta x_{n})/\|\delta x_{n}\|\to 0$.
\end{proof}

\begin{theorem}[Lorentzian signature from realizability]
Assume R, E, and~T. Then $Q$ is nondegenerate and indefinite, and after rescaling~$t$ so that $Q(dt,0)=dt^{2}$ and applying an invertible
linear change of variables,
\[
Q = dt^{2} - c_{\max}^{-2}\,dr^{2}
\]
for some $c_{\max}>0$. The null set consists of exactly two directions $dr/dt=\pm c_{\max}$.
\end{theorem}

\begin{proof}
\textbf{Step~1 (Null direction from~R).}
By~R, there exists a subsequence $v_{n_{k}}\to v\in S^{1}$ with
$v_{t}>0$, $v_{r}\neq 0$. By Lemma~4, $Q(v)\le 0$.

\textbf{Step~2 (Temporal direction positive).}
If $Q(1,0)\le 0$ then E gives $d(x;(\Delta t,0))=o(\Delta t)$,
contradicting~T. Hence $Q(1,0)>0$.

\textbf{Step~3 (Null vector by IVT).}
Define
\[
\gamma(s):=\frac{(1-s)(1,0)+s\,v}{\|(1-s)(1,0)+s\,v\|}, \qquad s\in[0,1].
\]
Since $v_{r}\neq 0$ (Assumption~R), the denominator is nonzero for all $s\in[0,1]$, so $\gamma$ is well-defined.
Since $v_{t}>0$, the numerator has strictly positive $t$-component throughout, so $\gamma$ is a continuous
path in $S^{1}\cap\{u_{t}>0\}$. Then $Q(\gamma(0))=Q(1,0)>0$ and $Q(\gamma(1))=Q(v)\le 0$, so by the
intermediate value theorem there exists~$s^{*}$ with $Q(\gamma(s^{*}))=0$. Set $w:=\gamma(s^{*})$; then $w_{t}>0$ and $w\neq(1,0)$, so $w_{r}\neq 0$.

\textbf{Step~4 (Nondegeneracy).}
Suppose $Q$ is degenerate, i.e.\
\[
\det\begin{pmatrix}a & \kappa \\ \kappa & b\end{pmatrix} = ab-\kappa^{2}=0,
\]
$Q\not\equiv 0$. Since $a=Q(1,0)>0$ (Step~2), the substitution $r'=r+(\kappa/a)t$ eliminates the cross
term: $Q=a\,dt^{2}+b'\,dr'^{2}$ with $ab'=0$, hence $b'=0$. Then $Q=a\,dt^{2}$ and $Q(w)=0$ forces
$w_{t}=0$, contradicting $w_{t}>0$ (Step~3).

\textbf{Step~5 (Signature and normalisation).}
By Step~2, $Q$ takes a positive value; by Step~3,
$Q$ vanishes on a nonzero vector. A nondegenerate quadratic form on~$\mathbb{R}^{2}$ with a nonzero null
vector cannot be definite; hence it is indefinite, and in two dimensions the signature is $(1,1)$.
Rescaling~$t$ so that $Q(dt,0)=dt^{2}$ and applying an invertible linear change of variables gives
\[
Q = dt^{2} - c_{\max}^{-2}\,dr^{2}.\qedhere
\]
\end{proof}

\begin{remark}[The speed $c_{\max}$]
The constant $c_{\max}$ is not an independent postulate but arises from the cost structure. It represents the maximal propagation speed, determined by the relative
coefficients of~$Q$. Note that $c_{\max}$ may vary from point to point: the present result is local. A
global constant $c_{\max}$ would follow from additional homogeneity assumptions on the cost function,
which are not required here.
\end{remark}

\begin{remark}[Null directions as zero-cost propagation]
On the two null directions $dr/dt=\pm c_{\max}$, the quadratic leading term vanishes:
$Q(\delta x)=0$. By Assumption~E, this implies $d(x;\delta x)=o(\|\delta x\|)$ --- the displacement cost
vanishes faster than the displacement magnitude. The null cone therefore separates
directions of strictly positive leading-order cost ($Q>0$, timelike) from directions along which
the cost function cannot sustain a positive quadratic term ($Q<0$, spacelike).
\end{remark}

\begin{remark}[Exclusion of the Galilean degeneracy]
The degenerate case $\det Q=0$ is excluded algebraically in Step~4 of Theorem~5. It is worth
noting the physical content of this exclusion. When $Q$ is degenerate with $a=Q(1,0)>0$, one
has $Q=a\,dt^{2}$ (after eliminating the cross term), so the null condition $Q=0$ forces $dt=0$.
A null displacement with $dt=0$ is instantaneous: it cannot represent the zero-cost boundary
of temporally persistent spatial propagation. In the normalised form, this corresponds to the
limit $c_{\max}\to\infty$ (equivalently $b\to 0$), in which the cone opens up and $Q$ reduces to
an absolute time increment. The finiteness of~$c_{\max}$ is therefore not an additional assumption
but a consequence of the requirement that zero-cost directions carry both $\delta t>0$ and $\delta r\neq 0$.
\end{remark}

\section{Extension to $(1{+}3)$ dimensions}

For the four-dimensional extension, let $U\subset\mathbb{R}^{4}$ be open, fix $x\in U$, and consider the chart
$(t,x^{1},x^{2},x^{3})$. The two-dimensional case is the minimal realisation: a single spatial direction
suffices. In physical spacetime, the condition must be imposed along all three independent
spatial directions.

\begin{definition}[R4 --- Three-directional zero-threshold realizability]
For each $i\in\{1,2,3\}$ there
exists a sequence $\delta x_{n}^{(i)}\to 0$ with $\delta t_{n}^{(i)}>0$ and $\delta x_{n}^{(i),i}\neq 0$, such that $d(x;\delta x_{n}^{(i)})/\|\delta x_{n}^{(i)}\|\to 0$, and
a subsequence of normalised directions converges to $v^{(i)}$ with $v_{t}^{(i)}>0$ and $v_{i}^{(i)}\neq 0$. The three
limiting directions $\{v^{(1)},v^{(2)},v^{(3)}\}$ are required to span a three-dimensional subspace~$W$, and
the restriction $Q|_{W}$ is required to be seminegative definite:
\[
Q(w)\le 0 \quad\text{for all } w\in W.
\]
\end{definition}

The seminegative condition on~$W$ strengthens the spanning requirement: it guarantees a
three-dimensional subspace on which $Q$ is seminegative definite, thereby ruling out signatures
such as $(2,2)$. This condition is strictly stronger than pointwise vanishing at the three directions~$v^{(i)}$: it requires $Q\le 0$ on all linear combinations of those directions, a coherence condition on
the spatial sector as a whole. The question of why physical spacetime satisfies this coherence
condition, and why it has precisely three spatial directions rather than more or fewer, lies outside
the scope of the present paper.

\begin{definition}[E4 --- Quadratic expansion in 4D]
There is a quadratic form $Q$ on~$\mathbb{R}^{4}$ such that
\[
d(x;\delta x)^{2} = \bigl(Q(\delta x)\bigr)_{+} + o\!\left(\|\delta x\|^{2}\right) \quad\text{as } \delta x\to 0 \text{ with } \delta t>0.
\]
\end{definition}

Assumption~T carries over unchanged.

\begin{theorem}[Lorentzian signature --- $(1{+}3)$D]
Assume R4, E4, and~T, and suppose that $Q$ is
nondegenerate. Then $\mathrm{Sig}(Q)=(1,3)$.
\end{theorem}

\begin{proof}
\textbf{Step~1 (Temporal direction positive).}
Assumption~T implies $Q(e_{t})>0$ by the same
argument as Step~2 of Theorem~5: if $Q(e_{t})\le 0$ then $d(x;(\Delta t,0,0,0))=o(\Delta t)$, contradicting~T.
Hence the positive inertia index of~$Q$ is at least~1.

\textbf{Step~2 (Seminegative spatial subspace).}
By Definition~7, the limiting directions
$\{v^{(1)},v^{(2)},v^{(3)}\}$ span a three-dimensional subspace $W\subset\mathbb{R}^{4}$ on which
\[
Q(w)\le 0 \quad\text{for all } w\in W.
\]
Thus $Q|_{W}$ is seminegative definite on a three-dimensional subspace, so the negative inertia index
of~$Q$ is at least~3.

\textbf{Step~3 (Inertia count).}
Since $Q$ is nondegenerate on~$\mathbb{R}^{4}$, the positive and negative inertia
indices sum to~4. The positive index is at least~1 (Step~1) and the negative index is at least~3
(Step~2); hence the only possibility is $\mathrm{Sig}(Q)=(1,3)$.
\end{proof}

\begin{remark}[Nondegeneracy in the $(1{+}3)$D theorem]
Theorem~9 assumes nondegeneracy of~$Q$ as a hypothesis, whereas Theorem~5 derives it. In four dimensions, ruling out degeneracy
from Assumptions~R4, E4, T alone requires additional input (e.g.\ realizability along a fourth
independent direction). The nondegeneracy hypothesis in Theorem~9 is therefore an explicit
scope condition, not a hidden assumption.
\end{remark}

\begin{remark}[Exclusion of non-Lorentzian signatures]
Under the hypotheses of Theorem~5 in $(1{+}1)$ dimensions, and of Theorem~9 in $(1{+}3)$ dimensions, every non-Lorentzian signature is excluded. In $(1{+}1)$ dimensions:
\begin{itemize}
\item[(i)] Positive-definite signature $\mathrm{Sig}(Q)=(2,0)$ is excluded by Assumption~R together with
Lemma~4: it yields a nonzero limiting direction~$v$ with $v_{t}>0$ and $Q(v)\le 0$, which is
impossible for a positive-definite form.
\item[(ii)] Negative-definite signature $\mathrm{Sig}(Q)=(0,2)$ is excluded by Assumption~T (Step~2 of Theorem~5): negative definiteness gives $Q(1,0)<0$, contradicting the positive temporal cost.
\item[(iii)] Degenerate forms ($\det Q=0$, $Q\not\equiv 0$) are excluded by Assumptions~R and~T jointly
(Step~4 of Theorem~5): degeneracy forces $Q=a\,dt^{2}$ with $a>0$; but Step~3 of the proof of
Theorem~5 yields~$w$ with $Q(w)=0$ and $w_{t}>0$, forcing $w_{t}=0$, a contradiction.
\end{itemize}
In $(1{+}3)$ dimensions (Theorem~9):
\begin{itemize}
\item[(iv)] The split signature $(2,2)$ is excluded by the seminegative subspace condition of~R4: that
condition forces the negative inertia index of~$Q$ to be at least~3, which is incompatible with $(2,2)$.
\end{itemize}
Lorentzian signature is therefore not merely compatible with the cost asymmetry --- it is the unique outcome.
\end{remark}
